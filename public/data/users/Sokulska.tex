\section*{Вступ}

В 60-70 роках минулого століття американські математики  Л.А.
Рубел та Б.А. Тейлор \cite{R-T} (див. також \cite{KA}, \cite{K-L})
розробили метод рядів Фур'є для цілих та мероморфних функцій. На
відміну від класичних підходів за функцію зростання вони
запропонували брати довільну додатну, неперервну, неспадну і
необмежену функцію $\lambda(r)$ і розглядати класи $\Lambda$
мероморфних функцій скінченного $\lambda$-типу, тобто таких, що
$T(r,f)\leq A \lambda(Br)$ для довільного $r>0$ і деяких додатних
сталих $A$ і $B$, де $T(r,f)$ --- характеристика Неванлінни функції
$f$.

Цей метод дозволив розв'язати низку важливих задач, зокрема, на
основі детального вивчення коефіцієнтів Фур'є
$$c_k(r,f)=\frac{1} {2\pi} \int \limits_0^{2\pi} e^{-ik\theta}
\log |f(re^{i\theta})|d\theta,\quad r>0, \quad  k\in \mathbb{Z},$$
повністю описати множину нулів цілих функцій з класів $\Lambda$.

В  даній роботі вводиться і досліджується аналог характеристики
Неванлінни для мероморфних в замиканні  півсмуги $S=\{\sigma+it:
\sigma>0, 0<t<2\pi \}$ функцій $f$ таких, що
$f(\sigma)=f(\sigma+2\pi i)$, $\sigma \geq 0$. За допомогою
методу, запропонованого Дж. Літтлвудом \cite{Li}, встановлюються
співвідношення для коефіцієнтів Фур'є мероморфних в $S$ функцій,
вводиться характеристика Неванлінни таких функцій, доводиться
критерій скінченності їх $\lambda$-типу.


\section{Формула Йенсена-Літтлвуда та характеристика Неванлінни мероморфної в півсмузі функції}
Нехай функція $f$ мероморфна в замиканні півсмуги
$S=\{\sigma+it:
\sigma>0, 0<t<2\pi \}$, не має ні
нулів, ні полюсів на $\partial S$, $f\not\equiv 0$ і, крім того,
$f(\sigma)=f(\sigma+2\pi i)$, $\sigma\geq0.$

Через $\{s_j\}$ позначимо послідовність нулів функції $f$ в $S$,
$s_j=\sigma_j+it_j$, через $\{p_j\}$ ---
послідовність її полюсів в $S$.
Через $S^\ast$ позначимо смугу $S$ з розрізами
$\{\tau\sigma_j+it_j\}$, $\{\tau \mbox{Re}p_j+i\mbox{Im}p_j\}$,
$1\leq\tau<\infty$. Нехай $s_0 \in S^\ast$ і вибране деяке
значення $\log f(s_0).$ Покладемо
\begin{eqnarray} {\label {eq_2.1}}
\log f(s)-\log f(s_0)=\int\limits_{s_0}^s
\frac{f'(\zeta)}{f(\zeta)}d\zeta,
\end{eqnarray}
де інтеграл береться по деякому шляху в $S^\ast \cup \partial S.$

Нехай $n(\eta, f)$ лічильна функція полюсів функції  $f$ у
прямокутнику
$R_\eta=\{\sigma+it: 0<\sigma \leq \eta,
0\leq t< 2\pi \},$
i
\begin{eqnarray*} {\label {eq_2.2}}
N(\sigma,f)=\int\limits_0^\sigma n(\eta,f)d\eta.
\end{eqnarray*}

Позначимо
\begin{eqnarray} {\label {eq_2.4}}
c_0(\sigma, f)=\frac{1}{2\pi}\int\limits_0^{2\pi} \log |f(\sigma
+it)|dt.
\end{eqnarray}


Аналог теореми Йенсена - Літтлвуда (\cite{Li}) формулюється
наступним чином.

\begin{lemma}
Нехай функція $f$ мероморфна в замиканні півсмуги $S$,
$f(\sigma)=f(\sigma+2\pi i),$ $\sigma\geq~0.$ Тоді
\begin{eqnarray} {\label {eq_2.15}}
N(\sigma, \frac{1} {f})-N(\sigma, f)=c_0(\sigma, f)-\frac{\sigma}
{{\sigma}_0} c_0({\sigma}_0, f)+ (\frac{\sigma}
{{\sigma}_0}-1)c_0(0, f), \quad \sigma\geq{\sigma}_0>0.
\end{eqnarray}
\end{lemma}

\begin{proof} Припустимо спочатку, що $f$ не має ні нулів, ні
полюсів на $\partial S.$
 Застосувавши принцип аргумента, отримаємо
$$\frac{1} {2\pi i} \int\limits_{\partial {R_\eta}}
\frac{f'(\zeta)}{f(\zeta)}d\zeta=n(\eta, \frac{1} {f})-n(\eta,
{f}).$$

 Межа
$\partial {R_\eta}$ складається з чотирьох відрізків, тому
\begin{equation}\label {eq_2.6}
\frac{1} {2\pi i}\left[\int\limits_{0}^{\eta}
\frac{f'(\sigma)}{f(\sigma)}d\sigma+i\int\limits_{0}^{2\pi}
\frac{f'(\eta+it)}{f(\eta+it)}dt-\int\limits_{0}^{\eta}
\frac{f'(\sigma+2\pi i)}{f(\sigma+2\pi
i)}d\sigma-i\int\limits_{0}^{2\pi} \frac{f'(it)}{f(it)}dt
\right]=n(\eta, \frac{1} {f})-n(\eta, {f}).
\end{equation}
 Оскільки $f$  має період  $2\pi i$,  то
$f'$ також має період $2\pi i$, тоді
$$\int\limits_{0}^{\eta}
\frac{f'(\sigma)}{f(\sigma)}d\sigma=\int\limits_{0}^{\eta}
\frac{f'(\sigma+2\pi i)}{f(\sigma+2\pi i)}d\sigma,$$
 і співвідношення (\ref {eq_2.6}) запишеться наступним чином
\begin{eqnarray} {\label {eq_2.7}}
\frac{1} {2\pi i}\left[i\int\limits_{0}^{2\pi}
\frac{f'(\eta+it)}{f(\eta+it)}dt-i\int\limits_{0}^{2\pi}
\frac{f'(it)}{f(it)}dt \right]=n(\eta, \frac{1} {f})-n(\eta, {f}).
\end{eqnarray}

Проінтегруємо рівність (\ref {eq_2.7})  по $\eta$ від $0$ до
$\sigma$
\begin{equation} {\label {eq_2.8}}
\int\limits_0^{\sigma} n(\eta, \frac{1}
{f})d\eta-\int\limits_0^{\sigma} n(\eta, {f})d\eta=\frac{1}
{2\pi}\int\limits_0^{\sigma} \int\limits_{0}^{2\pi}
\frac{f'(\eta+it)}{f(\eta+it)}dt d\eta-\frac{1}
{2\pi}\int\limits_0^{\sigma} \int\limits_{0}^{2\pi}
\frac{f'(it)}{f(it)}dt d\eta =
\mathcal{J}_1-\mathcal{J}_2.
\end{equation}

Враховуючи означення логарифма (\ref {eq_2.1}) функції $f$, другий
інтеграл співвідношення (\ref {eq_2.8}) запишемо у вигляді
\begin{eqnarray} {\label {eq_2.9}}
\mathcal{J}_2=\frac{1} {2\pi}\int\limits_0^{\sigma}
\int\limits_{0}^{2\pi} \frac{f'(it)}{f(it)}dt d\eta =
\frac{\sigma} {2\pi i}[\log f(2\pi i)-\log f(0)],
\end{eqnarray}

 Застосовуючи теорему
Фубіні до $\mathcal{J}_1$, отримаємо
\begin{eqnarray} {\label {eq_2.10}}
\mathcal{J}_1=\frac{1} {2\pi}\int\limits_{0}^{2\pi} dt
\int\limits_0^{\sigma} \frac{f'(\eta + it)}{f(\eta + it)} d\eta =
\frac{1} {2\pi}\int\limits_0^{2\pi}[\log f(\sigma+it)-\log
f(it)]dt,
\end{eqnarray}
 З врахуванням  (\ref {eq_2.9}) і (\ref {eq_2.10}), рівність (\ref {eq_2.8})
можна записати наступним чином
\begin{eqnarray} {\label {eq_2.11}}
N(\sigma, \frac{1} {f})-N(\sigma, f)=l_0(\sigma, f)-l_0(0, f)+i
\frac{\sigma} {2\pi}[\log f(2\pi i)-\log f(0)],
\end{eqnarray}
де $$l_0(\sigma, f)=\frac{1}{2\pi}\int\limits_0^{2\pi} \log
f(\sigma +it)dt.$$

 Взявши дійсні частини обох боків співвідношення (\ref {eq_2.11}), отримаємо
\begin{eqnarray*} {\label {eq_2.12}}
N(\sigma, \frac{1} {f})-N(\sigma, f)=\frac{1} {2\pi}
\int\limits_0^{2\pi}\log|f(\sigma+it)|dt-\frac{1} {2\pi}
\int\limits_0^{2\pi}\log|f(it)|dt+ \sigma C_f,
\end{eqnarray*}
де $ C_f=\frac{1}{2\pi} [\mbox{Arg}f(0)- \mbox{Arg}f(2\pi i)]$,
$\mbox{Arg}f=\mbox{Im}\log f$, тобто
\begin{eqnarray} {\label {eq_2.13}}
N(\sigma, \frac{1} {f})-N(\sigma, f)=c_0(\sigma, f)-c_0(0, f)+
\sigma C_f, \quad \sigma >0.
\end{eqnarray}

З припущення відсутності нулів та полюсів на $\partial S$  та з
ізольованості наявних випливає існування $\sigma_0>0$ такого, що
$f\neq 0, \infty $  при $\sigma\leq \sigma_0$. Тому з (\ref
{eq_2.13}) маємо
\begin{eqnarray} {\label {eq_2.14}}
0=\frac{1}{\sigma_0} c_0(\sigma_0, f)-\frac{1}{\sigma_0} c_0(0,
f)+ C_f, \quad \sigma >0.
\end{eqnarray}

Поділивши рівність (\ref {eq_2.13}) на $\sigma$ і віднявши від неї
(\ref {eq_2.14}), одержимо
\begin{equation*} {\label {eq_2.16}}
\frac{1} {\sigma}[N(\sigma, \frac{1} {f})-N(\sigma, f)]=\frac{1}
{\sigma}c_0(\sigma, f)-\frac{1} {{\sigma}_0} c_0({\sigma}_0, f)+
\left(\frac{1} {{\sigma}_0}-\frac{1} {\sigma}\right)c_0(0, f),  \quad
\sigma\geq{\sigma}_0>0,
\end{equation*}
 звідки й випливає (\ref {eq_2.15}).

При незначній модифікації доведення для випадку, коли нулі чи полюси
функції $f$ лежать на $\partial S$ отримаємо рівність (\ref{eq_2.15}).
\end{proof}

Позначимо
\begin{eqnarray} {\label {eq_2.5}}
m_0(\sigma, f)=\frac{1}{2\pi}\int\limits_0^{2\pi} {\log}^+
|f(\sigma +it)|dt,
\end{eqnarray}
де $x^+=\max\{x,0\}.$

\begin{definition}
Нехай функція $f$ мероморфна в $ \overline{S}$, $f\not\equiv 0$ i
$f(\sigma)=f(\sigma+2\pi i)$. Характеристикою Неванлінни функції
$f$ називається
\begin{equation} {\label {eq_2.17}}
T(\sigma, f)=m_0(\sigma, f)-\frac{\sigma} {{\sigma}_0}
m_0({\sigma}_0, f)+ \left(\frac{\sigma} {{\sigma}_0}-1\right)m_0(0, f)+
N(\sigma, f),
\quad \sigma\geq{\sigma}_0>0.
\end{equation}
\end{definition}

Властивості $T(\sigma, f)$ опишемо наступною теоремою.

\begin{theorem}{\label {th_2.1}} Нехай $f$ --- мероморфна в $\overline{S}$
функція, $f\not\equiv 0$ i $f(\sigma)=f(\sigma+2\pi i).$  Тоді її
характеристика Неванлінни $T(\sigma, f)$ володіє наступними
властивостями:
\begin{itemize}
\item[(i)] $T(\sigma, f)$ --- опукла відносно $\sigma,$ при $\sigma \geq
\sigma_0$;
\item[(ii)]$T({\sigma}_0, f)$ =0;
\item[(iii)]$T(\sigma, f)$ --- неспадна, при $\sigma \geq\sigma_0$;
\item[(iv)]$T(\sigma, f)=T(\sigma, \frac{1}{f}),$ при $\sigma \geq\sigma_0$.
%\item[(v)]$T(\sigma, f)$ - невід'ємна при $\sigma \geq \sigma_0$;
\end{itemize}
\end{theorem}

\begin{proof}
Нехай $\log z=\log|z|+i\arg_0 z$, $0\leq \arg_0 z<2 \pi$,
$z\neq 0.$ Тоді функція $F(z)=f(\log z)$ мероморфна при $|z|\geq 1$
і $f(\log|z|)=f(\log|z|+2\pi i).$

Зобразимо функцію $F$ у вигляді частки двох функцій $\frac{H}
{G}$, де $H$, $G$ --- голоморфні при $|z|\geq 1$ без спільних
нулів, i $h(s)=H(e^s)$, $g(s)=G(e^s).$
Тоді, застосовуючи (\ref {eq_2.15}) до $g$, маємо
$$N(\sigma, {f})=N(\sigma,\frac{1} {g})=c_0(\sigma,
g)-\frac{\sigma} {{\sigma}_0} c_0({\sigma}_0,g)+
\left(\frac{\sigma} {{\sigma}_0}-1\right)c_0(0,g) ,\nonumber
\\ \quad \sigma\geq{\sigma}_0>0.$$

Враховуючи, що $(a-b)^++b=max(a,b),$ характеристику (\ref{eq_2.17}) функції
$f=\frac{h}{g}$ запишемо у вигляді:
\begin{eqnarray} {\label {eq_2.21}}
T(\sigma, f)=\frac{1}
{2\pi}\int\limits_0^{2\pi}\max(\log|h({\sigma+it)}|,\log|g({\sigma+it}
)|)dt\nonumber
\\-\frac{\sigma}
{{\sigma}_0} \frac{1}
{2\pi}\int\limits_0^{2\pi}\max(\log|h({\sigma_0+it})|,\log|g({{\sigma}_0+it}
)|)dt\nonumber
\\
+\left(\frac{\sigma} {{\sigma}_0}-1\right)\frac{1}
{2\pi}\int\limits_0^{2\pi}\max(\log|h({it})|,\log|g({it})|)dt,
\quad \sigma\geq{\sigma}_0>0.
\end{eqnarray}

Позначимо $I(\sigma,f)=\frac{1}
{2\pi}\int\limits_0^{2\pi}\max(\log|h({\sigma+it})|,\log|g({\sigma+it})|)dt.$

Тоді $I(\sigma,F)=\frac{1}
{2\pi}\int\limits_0^{2\pi}\max(\log|H(e^{\sigma+it})|,\log|G(e^{\sigma+it})|)dt.$

Функція $u(z)=\max(\log|H(z)|,\log|G(z)|)$  субгармонійна при
$|z|\geq 1,$ тому $I(\sigma,F)$ опукла \cite[ст. 27]{He}, \cite[ст.
49]{BVH},  при $\sigma\geq 0$.

Згідно з (\ref {eq_2.21})
\begin{eqnarray} {\label {eq_2.18}}
T(\sigma, f)=I(\sigma,f)-\frac{\sigma} {{\sigma}_0}
I({\sigma}_0,f) +\left( \frac{\sigma} {{\sigma}_0}-1\right)I(0,f),
\quad \sigma\geq{\sigma}_0>0.
\end{eqnarray}

Оскільки $T(\sigma,f)$ є сумою опуклої відносно
$\sigma\geq\sigma_0$ функції $I(\sigma,f)$ та лінійної функції
$A\sigma+B$, то властивість $(i)$ виконується.
За властивістю опуклості $I(\sigma,f),$ при $0\leq\sigma$ маємо
 $I({\sigma}_0,f)\leq \frac{\sigma-{\sigma}_0} {\sigma} I(0,f)+
\frac{{\sigma}_0} {\sigma} I({\sigma},f).$
 Звідси, $ 0\leq I({\sigma},f)-\frac{\sigma} {{\sigma}_0} I({\sigma}_0,f)+
\left(\frac{\sigma}
{{\sigma}_0}-1\right) I(0,f).$
Таким чином, згідно з (\ref {eq_2.18}) ми отримали, що $T(\sigma,f)\geq 0.$
Якщо $\sigma={\sigma}_0$, то з (\ref {eq_2.18}) випливає, що
$T({\sigma}_0, f)=0$.

Оскільки $T(\sigma,f)$ опукла, відносно $\sigma,$ при
$\sigma\geq\sigma_0$, то за теоремою 1.6 із \cite [ст. 28]{He-Ke}
(див також \cite [ст. 11]{BVH}) в кожній точці вона має похідну
справа і
$$\lim\limits_{\sigma\rightarrow\sigma_0+0}
\frac{T(\sigma,f)-T(\sigma_0,f)}{\sigma-\sigma_0}=
\lim\limits_{\sigma\rightarrow\sigma_0+0}
\frac{T(\sigma,f)}{\sigma-\sigma_0}=T_+'(\sigma_0,f)\geq0.$$
Похідна справа опуклої функції за цією ж теоремою не спадає. Тому
$0\leq T_+'(\sigma_0,f)\leq T_+'(\sigma,f), \quad \sigma\geq
\sigma_0$. Отож, характеристика $T(\sigma,f)$ неспадна.

Покажемо, що виконується властивість $\textsl{(iv)}$.
Застосуємо рівність (\ref {eq_2.15}) до функції $\frac{1}{f}$:
\begin{equation*} {\label {eq_2.41}}
N(\sigma, f)- N(\sigma, \frac{1} {f})=c_0(\sigma, \frac{1}{f})-\frac{\sigma}
{{\sigma}_0} c_0({\sigma}_0, \frac{1}{f})+ (\frac{\sigma}
{{\sigma}_0}-1)c_0(0, \frac{1}{f}), \quad \sigma\geq{\sigma}_0>0.
\end{equation*}

Враховуючи (\ref {eq_2.4}) i властивість $\log
x=\log^+x-\log^+\frac{1}{x},$ для $x> 0$, отримаємо
\begin{eqnarray*} {\label {eq_2.42}}
N(\sigma, f)- N(\sigma, \frac{1} {f})=\frac{1}{2\pi}\int\limits_0^{2\pi} \log^+\frac{1}{|f(\sigma
+it)|}dt-\frac{1}{2\pi}\int\limits_0^{2\pi} \log^+|f(\sigma+it)|dt\nonumber
\\-\frac{\sigma} {{\sigma}_0} \frac{1}{2\pi}\int\limits_0^{2\pi} \log^+\frac{1}{|f(\sigma_0
+it)|}dt+\frac{\sigma} {{\sigma}_0} \frac{1}{2\pi}\int\limits_0^{2\pi} \log^+{|f(\sigma_0
+it)|}dt\nonumber
\\+(\frac{\sigma}
{{\sigma}_0}-1)\frac{1}{2\pi}\int\limits_0^{2\pi} \log^+\frac{1}{|f(it)|}dt-(\frac{\sigma}
{{\sigma}_0}-1)\frac{1}{2\pi}\int\limits_0^{2\pi} \log^+{|f(it)|}dt , \quad \sigma\geq{\sigma}_0>0.
\end{eqnarray*}
Звідси,
\begin{eqnarray} {\label {eq_2.43}}
\frac{1}{2\pi}\int\limits_0^{2\pi} \log^+|f(\sigma+it)|dt-\frac{\sigma} {{\sigma}_0} \frac{1}{2\pi}\int\limits_0^{2\pi} \log^+{|f(\sigma_0
+it)|}dt+(\frac{\sigma}
{{\sigma}_0}-1)\frac{1}{2\pi}\int\limits_0^{2\pi} \log^+{|f(it)|}dt\nonumber
\\+N(\sigma, f)\nonumber
\\= \frac{1}{2\pi}\int\limits_0^{2\pi} \log^+\frac{1}{|f(\sigma
+it)|}dt-\frac{\sigma} {{\sigma}_0} \frac{1}{2\pi}\int\limits_0^{2\pi} \log^+\frac{1}{|f(\sigma_0
+it)|}dt+(\frac{\sigma}
{{\sigma}_0}-1)\frac{1}{2\pi}\int\limits_0^{2\pi}
\log^+\frac{1}{|f(it)|}dt\nonumber
\\+ N(\sigma, \frac{1} {f}) , \quad \sigma\geq{\sigma}_0>0.
\end{eqnarray}
З рівності (\ref {eq_2.43}) та  означення 1.1 отримуємо
\textsl{(iv)}.
 Теорему доведено.
\end{proof}

\frac{e^{k\sigma}-1}{k} [\log{f(2\pi i)}-\log {f(0)}],\

\end{equation}

{2\pi}\int\limits_0^{2\pi}

 e^{-ks_j},\quad k\in\mathbb{Z}\setminus\{0\}.
\end{eqnarray}

 Оскільки $\partial R_\eta$ складається з 4-х відрізків, то (\ref
 {eq_3.3}) набуде вигляду:

\sum\limits_{s_j\in R_\eta} e^{-ks_j},\quad

\end{eqnarray}

Зважаючи на $2\pi i$-періодичність функцій $f(s)$ та $e^{-ks}$,

\section{Мероморфні в півсмузі функції скінченного $\lambda$ - типу}

\begin{definition}
Додатна неспадна неперервна і необмежена функція $\lambda(\sigma),$ при $\sigma
> \sigma_0$  називається \textit{функцією зростання}.
\end{definition}

\begin{definition}
Нехай $\lambda(\sigma)$ --- функція зростання, $f$ --- мероморфна функція
в $\overline{S}$, $f\not\equiv 0$ i $f(\sigma)=f(\sigma+2\pi i)$.
Функція \textit{$f$} називається \textit{функцією скінченного
$\lambda$-типу}, якщо $\exists~A,~B~>~0$ такі, що
    $ T(\sigma,f)\leq A \lambda(\sigma+B)$
для всіх $\sigma\geq \sigma_0$.
\end{definition}

Клас мероморфних в $S$ функцій скінченного $\lambda$-типу позначимо через
$\Lambda$, а клас голоморфних в $S$ функцій скінченного $\lambda$-типу
позначимо через $ \Lambda_H $.

\begin{theorem}
 Нехай функція $f$ голоморфна в замиканні півсмуги $S$ і така, що $f\not\equiv
0$, $ f( \sigma )=f( \sigma+it ),$
$\lambda (\sigma) $ --- функція зростання така, що
$\sigma=O(\lambda(\sigma)),\quad \sigma\rightarrow\infty$. Тоді
наступні твердження еквівалентні.
\begin{itemize}
 \item [(i)] $f\in\Lambda_H;$
 \item [(ii)]$ |c_k(\sigma,f)|\leq A \lambda(\sigma+B),$ для всіх $\sigma
>\sigma_0$ і деяких $A, B>0$, $k\in \mathbb{Z}$;
 \item [(iii)] $|c_k(\sigma,f)|\leq \frac{A \lambda(\sigma+B)}{|k|+1},$ для всіх
$\sigma >\sigma_0$ і деяких $A, B>0$, $k\in \mathbb{Z}$.
\end{itemize}
\end{theorem}

\begin{proof} Якщо виконується $(i)$, то $ T(\sigma,f)\leq A \lambda(\sigma+B),$ при деяких $A, B>0$ і всіх
$\sigma>\sigma_0$. Тоді згідно з (\ref {eq_2.4}), властивістю
$|\log x|=\log^+x+\log^+\frac{1}{x}$, де $x> 0$, і пунктом (iv)
теореми 1 отримаємо
$$|c_k(\sigma, f)|=\frac{1}{2\pi}\left|\int\limits_0^{2\pi} e^{-ikt}\log
|f(\sigma +it)|dt\right|\leq \frac{1}{2\pi}\int\limits_0^{2\pi} \left|\log
|f(\sigma +it)|\right|dt $$
$$\leq \frac{1}{2\pi}\int\limits_0^{2\pi} \log^+
|f(\sigma +it)|dt+\frac{1}{2\pi}\int\limits_0^{2\pi} \log^+
\frac{1}{|f(\sigma +it)|}dt$$
$$\leq
m_0(\sigma,f)+\left(\frac{\sigma}{\sigma_0}-1\right)m_0(0,f)+m_0(\sigma,\frac{1}
{f})+\left(\frac{\sigma}{\sigma_0}-1\right)m_0(0,\frac{1}{f})$$
$$-\left(\frac{\sigma}{\sigma_0}-1\right)\frac{1}{2\pi}\int\limits_0^{2\pi} |\log
|f(it)||dt\leq
2T(\sigma,f)+\frac{\sigma}{\sigma_0}\frac{1}{2\pi}\int\limits_0^{2\pi}
|\log |f(it)||dt$$
$$\leq 2A\lambda(\sigma+B)+C\sigma\leq
A_1\lambda(\sigma+B),\quad k\in \mathbb{Z},$$ де  $C$ стала,
$A_1=\max\{2A,C\}$.

Нехай виконується $(ii)$. Тоді з рівності  (\ref {eq_3.00}) маємо:

$$e^{-k}c_k(\sigma+1,f)-c_k(\sigma,f) =
-\frac{\overline{\alpha_{-k}(f)}} {2 k}
(e^{-k(\sigma+2)}-e^{-k\sigma}) +\frac{1}{2k}
\sum\limits_{\sigma<\sigma_j\leq\sigma+1} \left(
\frac{e^\sigma}{e^{s_j}} \right)^k$$
$$- \frac{e^{-k}} {2k}\sum\limits_{0<\sigma_j\leq\sigma+1} \left( \frac{e^{\overline{s_j}}} {e^{\sigma+1}}
\right)^{k}+\frac{e^{-k}} {2k}\sum\limits_{0<\sigma_j\leq\sigma} \left( \frac{e^{\overline{s_j}}} {e^{\sigma}}
\right)^{k}  ,\quad k\in\mathbb{N}. $$

Звідси

\begin{eqnarray*} {\label {eq_3.15}}
|c_k(\sigma,f)| \leq \frac{|c_k(\sigma+1,f)|}{e^k}
+\frac{|\overline{\alpha_{-k}(f)}|} {2 k}
+\frac{n(\sigma+1,\frac{1}{f})}{2k} +
\frac{n(\sigma+1,\frac{1}{f})}{2ke^k} +
\frac{n(\sigma,\frac{1}{f})}{2k} ,\quad k\in\mathbb{N}.
\end{eqnarray*}

Але, оскільки

$$N(\sigma+1,\frac{1} {f})\geq \int\limits_0^{\sigma+1} n(\eta,\frac{1}{f})d\eta\geq \int\limits_\sigma^{\sigma+1} n(\eta,\frac{1}{f})d\eta\geq
n(\sigma,\frac{1}{f}), $$
то
\begin{eqnarray*} {\label {eq_3.16}}
|c_k(\sigma,f)| \leq \frac{|c_k(\sigma+1,f)|}{e^k}
+\frac{|\overline{\alpha_{-k}(f)}|} {2 k}
+\frac{N(\sigma+1,\frac{1}{f})}{2k} +
\frac{N(\sigma+1,\frac{1}{f})}{2ke^k} +
\frac{N(\sigma,\frac{1}{f})}{2k} ,\quad k\in\mathbb{N}.
\end{eqnarray*}

Враховуючи (\ref {eq_2.15}), отримаємо таку
нерівність
\begin{eqnarray} {\label {eq_3.17}}
N(\sigma, \frac{1} {f})< |c_0(\sigma, f)|+ C_1\sigma, \quad
\sigma\geq{\sigma}_0>0,
\end{eqnarray}
де $C_1$ --- стала.

Використовуючи властивість $c_{-k}=\overline{c_k}$, умову $(ii)$  та   (\ref {eq_3.17}), отримуємо


\begin{eqnarray*} {\label {eq_3.18}}
|c_k(\sigma,f)| \leq \frac{A_1\lambda(\sigma+B_1)}{|k|+1}, \quad k\in\mathbb{Z},
\end{eqnarray*}
для всіх $\sigma\geq \sigma_0>0$  і деяких $A_1, B_1$, таких, що
$A_1=\max\{4A,3C_1\}, B_1=B+1.$


Припустимо, що виконується $(iii)$, тоді

$$T(\sigma, f)=m_0(\sigma, f)-\frac{\sigma} {{\sigma}_0}
m_0({\sigma}_0, f)+ (\frac{\sigma} {{\sigma}_0}-1)m_0(0, f)\leq \frac{1} {2\pi}
\int \limits_0^{2\pi} |\log|f(\sigma+it)||dt $$
$$+C\sigma\leq \left[\frac{1} {2\pi} \int \limits_0^{2\pi}\left| \log|f(\sigma+it)|\right|^2
dt\right]^{\frac{1}{2}}+C\sigma= \left[\sum
\limits_{k\in\mathbb{Z}}|c_k(\sigma,f)|^2\right]^\frac{1}{2}+C\sigma$$
$$\leq A \left[\sum\limits_{k\in\mathbb{Z}}
\frac{1}{(|k|+1)^2}\right]^{\frac{1}{2}}\lambda(\sigma+B)+C\sigma\leq
A_2\lambda(\sigma+B),$$
при всіх $\sigma\geq{\sigma}_0>0$ і для деяких сталих $C, A_2$, а це означає, що
$f\in\Lambda_H.$ Таким чином, з \textsl{(iii)} випливає \textsl{(i)}.
\end{proof}
