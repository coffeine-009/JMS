\section{Вступ}
 У роботах \cite{AL}, \cite{ACG1}, \cite{Z-cont}, \cite{Z-proc}
 досліджувались алгебри аналітичних функцій на банахових
 просторах, зокрема у \cite{ACG1} було вивчено деякі важливі
 властивості таких алгебр, у \cite{Z-proc} було описано спектр
 алгебри цілих функцій обмеженого типу на банаховому просторі.
 Ми досліджуємо алгебру $\W(X)$ функцій, які можна подати у вигляді
 абсолютно збіжного на обмежених підмножинах банахового простору
 ряду $(p,q)$-поліномів. На цій алгебрі існує природна структура
 алгебри Фреше. Також для алгебри $\W(X)$ вдається уза\-галь\-ни\-ти
 деякі ідеї та методи дослідження алгебр аналітичних функцій
 обмеженого типу.

 В даній роботі описано спектр
 алгебри $\W(X)$, а також введено і досліджено ди\-фе\-рен\-ціюван\-ня, пов'язані зі
 спект\-ром.
\section{Основні означення}
Нехай $X$ --- комплексний банахів простір. Нехай $p,q \in \Zplus$,
де через $\Zplus$ позначено множину цілих невід'ємних чисел.
 Функцію $A: X^{p+q}\to \C $, яка є лінійною по кожному з перших
 $p$ аргументів, антилінійною по кожному з останніх $q$
 аргументів, симетричною по перших $p$ аргументах і симетричною по
 останніх $q$ аргументах будемо називати $(p,q)$-лінійною
 симетричною формою. Простір неперервних $(p,q)$-лінійних
 симетричних форм із нормою
 $$
 \|A\| = \sup_{\|x_1\|\leq 1, \ldots, \|x_{p+q}\|\leq 1}
 |A(x_1,\ldots,x_{p+q})|
 $$
 позначимо $\mathcal{L}^s(^{p,q}X)$.

 Звуження $(p,q)$-лінійної симетричної форми на діагональ назвемо
 $(p,q)$-поліномом. Число $p$ назвемо степенем однорідності $P$ і
 позначимо $\deg_h P$, число $q$ --- степенем антиоднорідності $P$
 і позначимо $\deg_a P$. Простір неперервних $(p,q)$-поліномів із
 нормою
 $$
 \|P\| = \sup_{\|x\|\leq 1} |P(x)|
 $$
 позначимо $\mathcal{P}(^{p,q}X)$.

 Для $(p,q)$-полінома $P$ назвемо $(p,q)$-лінійну симетричну форму
 $A_P$, звуженням якої є $P$, формою, асоційованою з $P$. У
 \cite{mfat} побудовано так звану поляризаційну формулу, яка
 дозволяє за даним $(p,q)$-поліномом відновити асоційовану з ним
 форму:
 \begin{multline}\label{polarization formula}
 A_P(x_1,\ldots,x_{p+q}) = \frac{1}{(2q+1)2^{p+q}p!q!}
 \sum_{\varepsilon_1,\ldots,\varepsilon_{p+q}=\pm 1}
 \varepsilon_1 \varepsilon_2 \ldots \varepsilon_{p+q}
 \sum_{j=1}^{2q+1}  \alpha_j^{2q+1-p}
  \\
 \times
 P\Bigl( \alpha_j (\varepsilon_1x_1+\ldots+\varepsilon_px_p)
  +\varepsilon_{p+1}x_{p+1}+\ldots +
 \varepsilon_{p+q}x_{p+q}\Bigr),
 \end{multline}
 де $\alpha_j = \exp\left(\frac{2\pi(j-1) i }{2q+1}\right)$.

За допомогою поляризаційної формули в \cite{mfat} одержано оцінку
для норм $(p,q)$-полі\-но\-ма і асоційованої з ним
$(p,q)$-лінійної симетричної форми, яку називають
по\-ля\-ри\-за\-цій\-ною нерівністю:
 \begin{equation} \label{polarization inequality}
 \|P\| \leq \|A_P\| \leq c(p,q,X)\|P\|,
 \end{equation}
 де $c(p,q,X)$ --- поляризаційна константа, для якої справедлива подвійна
оцінка \linebreak
 $1\leq c(p,q,X)\leq \frac{(p+q)^{p+q}}{p!q!}$.
Зауважимо, що
 $
 \frac{(p+q)^{p+q}}{p!q!} = \frac{(p+q)^{p+q}}{(p+q)!}\cdot
 \frac{(p+q)!}{p!q!}.
 $
 Оскільки $\frac{(p+q)^{p+q}}{(p+q)!} \leq e^{p+q}$ і $\frac{(p+q)!}{p!q!} = C_{p+q}^p \leq
 2^{p+q}$, то $\frac{(p+q)^{p+q}}{p!q!} \leq (2e)^{p+q}$. Звідси
 одержимо оцінку
 \begin{equation} \label{degree estimate}
 \|P\| \leq \|A_P\| \leq (2e)^{p+q}\|P\|.
 \end{equation}
 Із означення норми $(p,q)$-лінійної симетричної форми і оцінки (\ref{degree estimate})
 отримуємо оцінку
 \begin{equation} \label{value estimate}
 |A_P(x_1,\ldots,x_{p+q})| \leq \|A_P\| \|x_1\| \ldots
 \|x_{p+q}\|.
 \end{equation}
 Із поляризаційної формули (\ref{polarization formula}) і поляризаційної нерівності
 (\ref{polarization inequality}) випливає, що простори
 $\mathcal{L}^s(^{p,q}X)$ і $\mathcal{P}(^{p,q}X)$ ізоморфні.

 \section{Алгебра типу Вінера функцій, породжених $(p,q)$-поліномами}
 Позначимо $\mathbb{Q}^+$ множину раціональних чисел, більших від
 нуля. Множину функцій $f: X \to \C$ таких, що $f = \sum_{m=0}^\infty \sum_{k=0}^m
 f_{k,m-k}$, де $f_{k,m-k} \in \mathcal{P}(^{k,m-k}X)$, і для
 кожного $r \in \mathbb{Q}^+$ ряд
 $$
  \|f\|_r = \sum_{m=0}^\infty \sum_{k=0}^m \sup_{\|x\|\leq r}|f_{k,m-k}(x)|
 $$
 є збіжним, будемо позначати $\W(X)$. Множина $\W(X)$ зі
 звичайними операціями мно\-жен\-ня на скаляр, додавання і множення
 функцій разом із системою норм $\{\|\cdot\|_r: \; r \in \mathbb{Q}^+\}$
 є алгеброю Фреше.

 Зауважимо, що оскільки $\sup_{\|x\|\leq r}|f_{k,m-k}(x)| = r^m
 \|f_{k,m-k}\|$, то
 $$
  \|f\|_r = \sum_{m=0}^\infty r^m \sum_{k=0}^m \|f_{k,m-k}\|.
 $$

 Для кожного лінійного неперервного функціонала $\phi$ на алгебрі
 $\W(X)$ існує число $r\in \mathbb{Q}^+$ таке, що $\phi$ ---
 неперервний відносно норми $\|\cdot\|_r$. Визначимо
 радіус-функцію $R:\W(X)' \to \R$, поклавши для $\phi \in \W(X)'$
 за $R(\phi)$ інфімум таких $r\in\mathbb{Q}^+$, що $\phi$ є
 неперервним відносно норми $\|\cdot\|_r$.

Звуження лінійного функціонала $\phi \in
 \W(X)'$ на простір $\mathcal{P}(^{p,q}X)$ будемо позначати
 $\pi_{p,q}(\phi)$. Наступні дві теореми встановлюють зв'язок між
 значенням радіус-функції і нормами звужень на простори
 неперервних $(p,q)$-поліномів для лі\-ній\-но\-го функціонала на
 $\W(X)$.

 \begin{theorem}(\cite{cherniv}) \label{radius function theorem 1}
 Для значення радіус-функції на лінійному неперервному функціоналі
 $\phi$ виконується рівність
 $$
  R(\phi) = \limsup_{m\to \infty} \max_{0 \leq k \leq m}
  \|\pi_{k,m-k}(\phi)\|^{1/m}.
 $$
 \end{theorem}

 \begin{theorem}(\cite{cherniv}) \label{radius function theorem 2}
 Нехай $\phi$ --- лінійний функціонал на алгебрі $\W(X)$. Якщо
 звуження $\phi$ на простори $\mathcal{P}(^{p,q}X)$ для всіх
 $p,q\in\Zplus$ неперервні і їх норми задовольняють оцінку
$\|\pi_{p,q}(\phi)\|\leq
 cs^{p+q}$ для деяких фіксованих $c,s>0$, то $\phi$ --- неперервний
 лінійний функ\-ціо\-нал і $R(\phi)\leq s$.
 \end{theorem}

 \section{Оператор зсуву}
Нехай $x\in X$. Визначимо оператор зсуву $T_x$ на алгебрі $\W(X)$
 наступним чином:
 $$
  (T_xf)(y)=f(x+y), \quad \textrm{де}\quad f\in \W(X),\; y\in X.
 $$
 \begin{theorem}
  При фіксованому $x\in X$ для функції $f\in \W(X)$ значення оператора
  зсуву $T_xf$ буде належати алгебрі $\W(X)$.
 \end{theorem}
 \begin{proof}
  Нехай $f=\sum_{m=0}^\infty \sum_{k=0}^m f_{k,m-k}$. Тоді
   $$
    (T_xf)(y)=f(x+y)=\sum_{m=0}^\infty \sum_{k=0}^m
    f_{k,m-k}(x+y).
   $$
  Відомо, що
   $$
    f_{k,m-k}(x+y)=\sum_{i=0}^k\sum_{j=0}^{m-k} C_k^iC_{m-k}^j
    A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j),
   $$
  де $A_{f_{k,m-k}}$ --- це $(k,m-k)$-лінійна симетрична форма,
  асоційована з $f_{k,m-k}$, а через $A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j)$
  позначено $A_{f_{k,m-k}}(\underbrace{x,\ldots,x}_{k-i},\underbrace{y,\ldots,y}_i;\underbrace{x,\ldots,x}_{m-k-j},\underbrace{y,\ldots,y}_j).$

 Звідси
   \begin{multline*}
    (T_xf)(y)=\sum_{m=0}^\infty \sum_{k=0}^m \sum_{i=0}^k
    \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j)
    \\ = \sum_{i=0}^\infty \sum_{j=0}^\infty \sum_{m=i+j}^\infty
    \sum_{k=i}^{m-j} C_k^i C_{m-k}^j
    A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j).
   \end{multline*}

  Переконаємося, що для кожного $r > 0$ існує скінченна норма
  $\|T_xf\|_r$. За озна\-чен\-ням,
  $$
   \|T_xf\|_r = \sum_{i=0}^\infty \sum_{j=0}^\infty
    \sup_{\|y\|\leq r}
     \bigl|
      \sum_{m=i+j}^\infty \sum_{k=i}^{m-j}
       C_k^i C_{m-k}^j A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j)
     \bigr|.
  $$
  Звідси
  \begin{multline*}
   \|T_x f\|_r \leq \sum_{i=0}^\infty \sum_{j=0}^\infty
   \sum_{m=i+j}^\infty \sum_{k=i}^{m-j} C_k^i C_{m-k}^j
    \sup_{\|y\|\leq r}
    \bigl|
     A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j)
    \bigr|
      \\ =
      \sum_{m=0}^\infty \sum_{k=0}^m \sum_{i=0}^k
      \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
       \sup_{\|y\|\leq r}
        \bigl|
        A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j)
        \bigr|.
  \end{multline*}
 Згідно із оцінкою (\ref{value estimate}),
   $$
    |A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j)|\leq
     (2e)^m\|f_{k,m-k}\| \|x\|^{m-i-j} \|y\|^{i+j}.
   $$
 Звідси
   $$
    \sup_{\|y\|\leq r} |A_{f_{k,m-k}}(x^{k-i},y^i;x^{m-k-j},y^j)|
    \leq
     (2e)^m\|f_{k,m-k}\| \|x\|^{m-i-j} r^{i+j}.
   $$
 Тому
   \begin{multline*}
    \|T_x f\|_r \leq \sum_{m=0}^\infty \sum_{k=0}^m \sum_{i=0}^k
    \sum_{j=0}^{m-k} C_k^i C_{m-k}^j (2e)^m
    \|f_{k,m-k}\|\|x\|^{m-i-j}r^{i+j}
    \\=
     \sum_{m=0}^\infty (2e)^m \sum_{k=0}^m \|f_{k,m-k}\|
     \sum_{i=0}^k C_k^i r^i \|x\|^{k-i} \sum_{j=0}^{m-k} C_{m-k}^j
     r^j \|x\|^{m-k-j}
    \\=
     \sum_{m=0}^\infty (2e)^m \sum_{k=0}^m
     \|f_{k,m-k}\|(r+\|x\|)^k (r+\|x\|)^{m-k}
    \\=
     \sum_{m=0}^\infty (2e(r+\|x\|))^m \sum_{k=0}^m
     \|f_{k,m-k}\|=\|f\|_{2e(r+\|x\|)}.
   \end{multline*}
 Отже, $\|T_xf\|_r\leq \|f\|_{2e(r+\|x\|)}$. Оскільки $f\in \W(X)$,
 то $\|f\|_{2e(r+\|x\|)}$ --- скінченна, тому скінченною для
 довільного $r$ буде $\|T_xf\|_r$. Звідси, $T_xf\in \W(X)$.
 \end{proof}

 \begin{theorem}
  Нехай $\phi$ --- лінійний неперервний функціонал на $\W(X)$, $f$ ---
  деяка фіксована функція, яка належить алгебрі $\W(X)$. Тоді функція
  $x \mapsto \phi(T_xf)$ належить алгебрі $\W(X)$.
 \end{theorem}
 \begin{proof}
  Нехай $f=\sum_{m=0}^\infty \sum_{k=0}^m f_{k,m-k}$. Тоді
  $$
   (T_xf)(y)=\sum_{m=0}^\infty \sum_{k=0}^m \sum_{i=0}^k
   \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
   A_{f_{k,m-k}}(x^i,y^{k-i};x^j,y^{m-k-j}).
  $$
  Нехай $P_{x,f_{k,m-k}}^{k-i,m-k-j}: y \longmapsto
  A_{f_{k,m-k}}(x^i,y^{k-i};x^j,y^{m-k-j})$. Очевидно, що
  $P_{x,f_{k,m-k}}^{k-i,m-k-j}$ ---\linebreak $(k-i,m-k-j)$-поліномом. Тепер
  $$
   T_xf=\sum_{m=0}^\infty \sum_{k=0}^m \sum_{i=0}^k
   \sum_{j=0}^{m-k} C_k^i C_{m-k}^j P_{x,f_{k,m-k}}^{k-i,m-k-j}
  $$
  і
  \begin{multline*}
   \phi(T_xf)=\sum_{m=0}^\infty \sum_{k=0}^m \sum_{i=0}^k
   \sum_{j=0}^{m-k} C_k^i C_{m-k}^j \pi_{k-i,m-k-j}(\phi)(P_{x,f_{k,m-k}}^{k-i,m-k-j})
   \\=
   \sum_{i=0}^\infty \sum_{j=0}^\infty \sum_{m=i+j}^\infty
   \sum_{k=i}^{m-j} C_k^i C_{m-k}^j
   \pi_{k-i,m-k-j}(\phi)(P_{x,f_{k,m-k}}^{k-i,m-k-j}).
  \end{multline*}
  Оскільки відображення $x \mapsto
  \pi_{k-i,m-k-j}(\phi)(P_{x,f_{k,m-k}}^{k-i,m-k-j})$ є
  $(i,j)$-поліномом, то ві\-до\-бра\-жен\-ня $x \mapsto \sum_{m=i+j}^\infty
   \sum_{k=i}^{m-j} C_k^i C_{m-k}^j
   \pi_{k-i,m-k-j}(\phi)(P_{x,f_{k,m-k}}^{k-i,m-k-j})$ також є
   $(i,j)$-поліномом. Отже, ві\-до\-бра\-жен\-ня $x \mapsto \phi(T_xf)$
   зображається у вигляді ряду $(p,q)$-поліномів. Доведемо, що
   для довільного $r>0 $ буде $\|x \mapsto\phi(T_xf)\|_r<\infty$. За
   означенням,
   $$
    \|x \mapsto \phi(T_xf)\|_r=\sum_{i=0}^\infty
    \sum_{j=0}^\infty \sup_{\|x\|\leq r}
    \bigl|
     \sum_{m=i+j}^\infty \sum_{k=i}^{m-j} C_k^i C_{m-k}^j
     \phi_{k-i,m-k-j}(P_{x,f_{k,m-k}}^{k-i,m-k-j})
    \bigr|.
   $$
   Звідси
   $$
    \|x \mapsto \phi(T_xf)\|_r \leq \sum_{i=0}^\infty \sum_{j=0}^\infty
    \sum_{m=i+j}^\infty \sum_{k=i}^{m-j} C_k^i C_{m-k}^j \sup_{\|x\|\leq r}
    |
     \pi_{k-i,m-k-j}(\phi)(P_{x,f_{k,m-k}}^{k-i,m-k-j})
    |.
   $$
   Оскільки $| \pi_{k-i,m-k-j}(\phi)(P_{x,f_{k,m-k}}^{k-i,m-k-j})|\leq
   \|\pi_{k-i,m-k-j}(\phi)\| \|P_{x,f_{k,m-k}}^{k-i,m-k-j}\|$ і
   \begin{multline*}
    \|P_{x,f_{k,m-k}}^{k-i,m-k-j}\|=\sup_{\|y\|\leq 1}
    | P_{x,f_{k,m-k}}^{k-i,m-k-j}(y)|
    =\sup_{\|y\|\leq 1} |A_{f_{k,m-k}}(x^i,y^{k-i};x^j,y^{m-k-j})|
    \\ \leq \sup_{\|y\|\leq 1} (2e)^m
    \|f_{k,m-k}\|\|x\|^i\|y\|^{k-i}\|x\|^j \|y\|^{m-k-j}=(2e)^m
    \|f_{k,m-k}\|\|x\|^{i+j},
   \end{multline*}
   то
   \begin{multline*}
    \sup_{\|x\|\leq r}
    |
     \pi_{k-i,m-k-j}(\phi)(P_{x,f_{k,m-k}}^{k-i,m-k-j})
    |\leq \sup_{\|x\|\leq r} (2e)^m \|\pi_{k-i,m-k-j}(\phi)\|
    \|f_{k,m-k}\|\|x\|^{i+j}
    \\=(2e)^m
    \|\pi_{k-i,m-k-j}(\phi)\|
    \|f_{k,m-k}\|r^{i+j}.
   \end{multline*}
   Оскільки $\phi$ --- неперервний лінійний функціонал, то
   $R(\phi)<\infty$. Візьмемо деяке число $s>R(\phi)$. Тоді існує
   таке число $c>1$, що $\|\pi_{k,m-k}(\phi)\|\leq cs^m$. Звідси
   \begin{multline*}
   \|x \mapsto \phi(T_xf)\|_r \leq \sum_{m=0}^\infty \sum_{k=0}^m
   \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
   cs^{m-i-j}(2e)^m\|f_{k,m-k}\|r^{i+j}
   \\=
   c \sum_{m=0}^\infty (2e)^m \sum_{k=0}^m \|f_{k,m-k}\|
   \sum_{i=0}^k C_k^is^{k-i}r^i \sum_{j=0}^{m-k} C_{m-k}^j
   s^{m-k-j}r^j
   \\ = c \sum_{m=0}^\infty (2e)^m \sum_{k=0}^m \|f_{k,m-k}\|
   (r+s)^m  \\ = c \sum_{m=0}^\infty (2e(r+s))^m \sum_{k=0}^m
   \|f_{k,m-k}\| =  c \|f\|_{2e(r+s)}<\infty.
   \end{multline*}
 Отже, $(x\mapsto \phi(T_xf))\in \W(X)$.
  \end{proof}

 \section{Згортка лінійних функціоналів}
 Для лінійних неперервних функціоналів $\phi, \theta \in \W(X)'$
 визначимо згортку $\phi * \theta$ як функцію на $\W(X)$ формулою
 $$
  (\phi * \theta)(f)=\phi( x \mapsto \theta ( y \mapsto
  (T_xf)(y))),\quad f \in \W(X).
 $$
 Очевидно, що $\phi * \theta$ є лінійним функціоналом. Доведемо
 неперервність.
 \begin{theorem}
  Для лінійних неперервних функціоналів $\phi, \theta \in \W(X)'$
  згортка $\phi * \theta$ є лінійним неперервним функціоналом.
  При цьому $R(\phi * \theta) \leq  2e (R(\phi)+R(\theta))$.
 \end{theorem}
 \begin{proof}
  Оцінимо норми звужень згортки $\phi * \theta$ на простори
  $(k,m-k)$-поліномів, де $m \geq 0$, $ 0 \leq k \leq m $. Для
  цього оцінимо значення $\phi * \theta$ на $(k,m-k)$-поліномі
  $f_{k,m-k}$.
  \begin{multline*}
   (\phi * \theta)(f_{k,m-k})
   = \phi( x \mapsto \theta ( y \mapsto (T_x f_{k,m-k})(y) ) )
   = \phi( x \mapsto \theta ( y \mapsto f_{k,m-k}(x+y) ) )
   \\= \phi\Bigl( x \mapsto \theta \Bigl( y \mapsto
    \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    A_{f_{k,m-k}}( x^i, y^{k-i}; x^j, y^{m-k-j} ) \Bigr) \Bigr)
    \\
   = \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    \phi( x \mapsto \theta ( y \mapsto
    A_{f_{k,m-k}}( x^i, y^{k-i}; x^j, y^{m-k-j} ) ) ).
  \end{multline*}
  Оцінимо значення $(\phi * \theta)(f_{k,m-k})$ по модулю.
  \begin{multline*}
   |(\phi * \theta)(f_{k,m-k})|
   \leq \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
   |\phi( x \mapsto \theta ( y \mapsto
    A_{f_{k,m-k}}( x^i, y^{k-i}; x^j, y^{m-k-j} ) ) )|
   \\
   \leq \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    \|\pi_{i,j}(\phi)\| \sup_{\|x\|\leq 1}
    |\theta ( y \mapsto
    A_{f_{k,m-k}}( x^i, y^{k-i}; x^j, y^{m-k-j} ) )|
   \\
   \leq \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    \|\pi_{i,j}(\phi)\| \sup_{\|x\|\leq 1}
    \|\pi_{k-i,m-k-j}(\theta)\| \sup_{\|y\|\leq 1}
    |A_{f_{k,m-k}}( x^i, y^{k-i}; x^j, y^{m-k-j} )|
   \\
   \leq \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    \|\pi_{i,j}(\phi)\| \|\pi_{k-i,m-k-j}(\theta)\|
    \sup_{\|x\|\leq 1} \sup_{\|y\|\leq 1}
    \|A_{f_{k,m-k}}\| \|x\|^i \|y\|^{k-i} \|x\|^j \|y\|^{m-k-j}
   \\
   = \|A_{f_{k,m-k}}\|
    \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    \|\pi_{i,j}(\phi)\| \|\pi_{k-i,m-k-j}(\theta)\|
   \\
   \leq (2e)^m \|f_{k,m-k}\|
    \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    \|\pi_{i,j}(\phi)\| \|\pi_{k-i,m-k-j}(\theta)\|.
  \end{multline*}
  Тепер оцінимо норму лінійного функціонала $\pi_{k,m-k}(\phi *
  \theta)$.
  \begin{multline*}
   \|\pi_{k,m-k}( \phi * \theta )\|
   = \sup_{\|f_{k,m-k}\|\leq 1} |\pi_{k,m-k}( \phi * \theta )(f_{k,m-k})|
   \\
   \leq
    (2e)^m \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
   \|\pi_{i,j}(\phi)\| \|\pi_{k-i,m-k-j}(\theta)\|.
  \end{multline*}
  Візьмемо числа $s_1>R(\phi)$ і $s_2>R(\theta)$. Тоді існують
  такі числа $c_1,c_2>0$, що для довільних $p,q \in \mathbb{Z}_+$
  буде $\|\pi_{p,q}(\phi)\|\leq c_1 s_1^{p+q}$ і $\|\pi_{p,q}(\theta)\|\leq c_2
  s_2^{p+q}$. Використаємо ці оцінки.
  \begin{multline*}
   \|\pi_{k,m-k}( \phi * \theta )\|
   \leq (2e)^m \sum_{i=0}^k \sum_{j=0}^{m-k} C_k^i C_{m-k}^j
    c_1 s_1^{i+j} c_2 s_2^{k-i+m-k-j}
   \\
   = c_1 c_2 (2e)^m
    \sum_{i=0}^k C_k^i s_1^i s_2^{k-i}
    \sum_{j=0}^{m-k} C_{m-k}^j s_1^j s_2^{m-k-j}
   = c_1 c_2 (2e)^m (s_1+s_2)^m.
  \end{multline*}
  Позначимо $c=c_1 c_2$, $s=2e(s_1+s_2)$. Тоді для норм звужень
  згортки на простори\linebreak $(k,m-k)$-поліномів маємо оцінку
  $\|\pi_{k,m-k}( \phi * \theta )\|\leq cs^m$. За теоремою \ref{radius function theorem 2}
  згортка $\phi * \theta$ є не\-пе\-рерв\-ним лінійним функціоналом на
  алгебрі $\W(X)$. Для радіус-функції маємо оцінку
  $
   R(\phi * \theta)\leq s = 2e(s_1+s_2).
  $
  Оскільки $s_1$ і $s_2$ вибрані довільно, то
  $$
   R(\phi * \theta)\leq 2e(R(\phi)+R(\theta)).
  $$
  \end{proof}

  Доведемо ще одну властивість згортки.
 \begin{theorem}
  Нехай $\phi, \theta$ --- лінійні неперервні функціонали на
  алгебрі $\W(X)$. Якщо $\phi, \theta$ є характерами, то
  $\phi * \theta$ також є характером.
 \end{theorem}
 \begin{proof}
  Перевіримо, що для довільних функцій $f,g \in \W(X)$ виконується
  рівність
  $$
   (\phi * \theta)(fg) = (\phi * \theta)(f) (\phi * \theta)(g).
  $$
  За означенням,
  $
   (\phi * \theta)(fg) = \phi(x \mapsto \theta(y \mapsto
   (fg)(x+y))).
  $
  Оскільки $\phi, \theta$ є характерами, то
  $$
   \theta(y \mapsto (fg)(x+y))
   = \theta(y \mapsto f(x+y)g(x+y))
   = \theta(y \mapsto f(x+y))
     \theta(y \mapsto g(x+y))
  $$
  і
  \begin{multline*}
   (\phi * \theta)(f g)
   = \phi\Bigl(x \mapsto
      \theta(y \mapsto f(x+y))
      \theta(y \mapsto g(x+y))\Bigr)
   \\
   = \phi\Bigl(x \mapsto
      \theta(y \mapsto f(x+y))\Bigr)
     \phi\Bigl(x \mapsto
      \theta(y \mapsto g(x+y))\Bigr)
   = (\phi * \theta)(f)(\phi * \theta)(g).
  \end{multline*}

\end{proof}

 Для скінченної послідовності $\{\phi_k\}_{k=1}^n \subset \W(X)'$
 згортку $\phi_1 * \ldots * \phi_n$ будемо позначати $\con_{k=1}^n
 \phi_k$.

 \section{Спектр алгебри $\W(X)$}
Нехай $(n,l)\in \mathbb{Z}_+^2$. Визначимо множини
 $\overline{\Lambda}_{n,l}, \Lambda_{n,l} \subset \mathbb{Z}_+^2$
 наступним чином
 $$
  \overline{\Lambda}_{n,l} = \{ (i,j):\; 0\leq i \leq n,\; 0\leq j \leq l
  \},
 $$
 $$
  \Lambda_{n,l}=\overline{\Lambda}_{n,l}\setminus \{(n,l)\}.
 $$

 Нехай $\W_{\Lambda_{n,l}}(X)$ --- замикання підалгебри алгебри
 $\W(X)$, породженої $(p,q)$-по\-лі\-но\-ма\-ми, для яких $(p,q) \in \Lambda_{n,l}$
 і нехай $\W_{\overline{\Lambda}_{n,l}}(X)$ --- замикання підалгебри алгебри
 $\W(X)$, по\-ро\-дже\-ної $(p,q)$-поліномами, для яких $(p,q) \in
 \overline{\Lambda}_{n,l}$.

 У \cite{cherniv} доведено таку теорему:
 \begin{theorem} \label{thm5.1}
 Нехай $\phi$ --- лінійний неперервний функціонал на $\W(X)$ такий,
 що $\phi(P) = 0$ для кожного $P \in \mathcal{P}(^{n,l}X) \cap
 \W_{\Lambda_{n,l}}(X)$,
 але при цьому $\pi_{n,l}(\phi) \neq 0$. Тоді існує неперервний
 комплекснозначний гомоморфізм $\psi$ на $\W(X)$ такий, що
 $\pi_{p,q}(\psi)=0$, якщо $(p,q)\in \Lambda_{n,l}$, і
 $\pi_{n,l}(\psi)=\pi_{n,l}(\phi)$. Радіус-функція
 $$
 R(\psi)
 \leq e
 \frac{
       n+l
      }
      {
       n^{{n}/{(n+l)}}
       l^{{l}/{(n+l)}}
      }
 \|\pi_{n,l}(\phi)\|^{1/(n+l)}
.$$
 \end{theorem}
Із того, що $\psi$ --- характер і $\pi_{p,q}(\psi)=0$ для $(p,q)\in
\Lambda_{n,l}$ випливає
\begin{corollary} \label{cor}
$\pi_{p,q}(\psi)$ не дорівнює нулю тільки якщо існує $k\in \Zplus$
таке, що $p=kn$ і $q=kl$.
\end{corollary}

 Побудуємо функцію $\varkappa: \mathbb{Z}_+^2 \to \mathbb{Z}_+$,
 поклавши $\varkappa(i,j) = \frac{(i+j)(i+j+1)}{2} + i$ для
 $(i,j) \in \mathbb{Z}_+^2$. Можна перевірити, що $\varkappa$ є
 бієкцією. Позначимо через $\gamma$ обернене відображення до
 $\varkappa$.

 Нехай $I_k$ буде мінімальний замкнутий ідеал алгебри $\W(X)$,
 породжений всіма не\-пе\-рерв\-ними $(p,q)$-поліномами, для яких
 $0< \varkappa(p,q) \leq k$.

 Множину характерів алгебри $\W(X)$ будемо позначати $M_{\W(X)}$.

 Нехай для $k \geq 1$
 $$
  \Phi_k = \{ \phi \in M_{\W(X)}: \quad \ker \phi \supset I_k \}.
 $$
 Також покладемо $\Phi_0 = M_{\W(X)}$.

 Доведемо деякі допоміжні результати.
 \begin{theorem}\label{thm8.1}
   Якщо для пари $(n,l)\neq (0,0)$ алгебра $\W_{\Lambda_{n,l}}(X)$
  не співпадає з алгеброю $\W_{\overline{\Lambda}_{n,l}}(X)$, то
  існує такий характер $\psi \in \Phi_{\varkappa (n,l)-1}$, що
  $\psi \notin \Phi_{\varkappa (n,l)}$.
 \end{theorem}
 \begin{proof}
   Нехай $P \in \mathcal{P}(^{n,l}X)$ і $P \notin
  \W_{\Lambda_{n,l}}(X)$. Оскільки алгебра $\W_{\Lambda_{n,l}}(X)$
  є замк\-ну\-тим підпростором алгебри $\W(X)$, то за теоремою
  Гана-Банаха існує лінійний функ\-ціо\-нал $\phi \in \W(X)'$ такий, що
  $\phi(f)=0$ для кожної функції $f \in \W_{\Lambda_{n,l}}(X)$ і $\phi(P) \neq 0$.
    За тео\-ре\-мою \ref{thm5.1} існує характер $\psi$ на алгебрі $\W(X)$
  такий, що $\psi_{p,q}=0$, якщо $(p,q)\in \Lambda_{n,l}$ і
  $\psi_{n,l}=\phi_{n,l}$. Неважко переконатись, що для жодного значення $k\in
\N$ пара
  $(kn,kl)$ не на\-ле\-жить множині $\{ (i,j): 0<\varkappa(i,j)\leq \varkappa(n,l)-1
  \}$, тому за наслідком \ref{cor} $\ker \psi \supset I_{\varkappa(n,l)-1}$ і $\psi \in \Phi_{\varkappa(n,l)-1}$. Оскільки
  $\psi(P)\neq 0$, то $\psi \notin \Phi_{\varkappa(n,l)}$.
 \end{proof}

 \begin{theorem}\label{thm8.2}
  Якщо $\phi, \psi \in M_{\W(X)}$, $\phi, \psi \neq 0$ і $\psi \in
  \Phi_{\varkappa(n,l)-1}$,
  то $(\phi * \psi)(P)=\phi(P)+\psi(P)$ для довільного
  $P\in\mathcal{P}(^{n,l}X)$.
 \end{theorem}
 \begin{proof}
  За означенням, $(\phi * \psi)(P) = \phi(x\mapsto \psi(y\mapsto
  (T_xP)(y)))$. Оскільки
  $$
   (T_xP)(y) = P(x+y)
   = \sum_{i=0}^n \sum_{j=0}^l C_n^i C_l^j
    A_P(x^i, y^{n-i}; x^j, y^{l-j}),
  $$
  то
  $$
   \psi(y\mapsto (T_xP)(y))
   = \sum_{i=0}^n \sum_{j=0}^l C_n^i C_l^j
   \psi(y\mapsto A_P(x^i, y^{n-i}; x^j, y^{l-j})).
  $$
  Оскільки $\psi \in \Phi_{\varkappa(n,l)-1}$, то
  $\pi_{\gamma(k)}(\psi)=0$ для $1\leq k \leq \varkappa(n,l)-1$.
    Звідси випливає, що $\pi_{n-i,l-j}(\psi)=0$ при
  $(n-i,l-j)\in \Lambda_{n,l}\setminus \{(0,0)\}$,
  тому
  \begin{multline*}
   \psi(y\mapsto (T_xP)(y))
   = \psi(y \mapsto A_P(x^n; x^l)) +
     \psi(y \mapsto A_P(y^n; y^l))
   \\
   = \psi(y \mapsto P(x)) +
     \psi(y \mapsto P(y))
   = P(x)\psi(y\mapsto 1)+\psi(P).
  \end{multline*}
  Оскільки $\psi$ --- ненульовий характер, то $\psi(y\mapsto 1)=1$.
  Тому
  $$
   \psi(y\mapsto (T_xP)(y))=P(x)+\psi(P).
  $$
  Звідси
  \begin{multline*}
   (\phi * \psi)(P)
   = \phi(x\mapsto P(x)+\psi(P))
   = \phi(x\mapsto P(x))+\phi(x\mapsto \psi(P))
   \\
   = \phi(P)+\psi(P)\phi(x\mapsto 1)
   = \phi(P)+\psi(P).
  \end{multline*}
\end{proof}

 \begin{theorem} \label{thm8.3}
  Якщо $P\in\mathcal{P}(^{n,l}X)$ і послідовність $\{\phi_j\}_{j=1}^{+\infty}$
  така, що $\phi_j \in \Phi_{j-1}$, то для довільного
  $m>\varkappa(n,l)$ буде $(\con_{j=1}^m \phi_j)(P) = (\con_{j=1}^{\varkappa(n,l)}
  \phi_j)(P)$.
 \end{theorem}
 \begin{proof}
  Оскільки $\phi_m \in \Phi_{m-1}\subset \Phi_{\varkappa(n,l)-1}$,
  то за теоремою \ref{thm8.2}
  $$
   (\con_{j=1}^m \phi_j)(P)
   = ((\con_{j=1}^{m-1} \phi_j)*\phi_m)(P)
   = (\con_{j=1}^{m-1} \phi_j)(P)+\phi_m(P).
  $$
 З того, що $\phi_m \in \Phi_{m-1}\subset \Phi_{\varkappa(n,l)}$
 випливає, що $\phi_m(P)=0$. Тому
 $
 (\con_{j=1}^m \phi_j)(P)
 =(\con_{j=1}^{m-1} \phi_j)(P).
 $
 Продовжуючи аналогічно, одержимо
  $$
   (\con_{j=1}^{m-1} \phi_j)(P)
   = (\con_{j=1}^{m-2} \phi_j)(P)
   = \ldots
   = (\con_{j=1}^{\varkappa(n,l)}
  \phi_j)(P).
  $$
 \end{proof}

 Для послідовності $\{\phi_n\}_{n=1}^\infty \subset M_{\W(X)}$, $\phi_n \in \Phi_{n-1}$
 нескінченною згорткою будемо вважати такий лінійний мультиплікативний
 функціонал на алгебрі функцій, породжених скін\-чен\-ни\-ми лінійними комбінаціями і добутками $(p,q)$-поліномів,
 що $(\con_{n=1}^\infty \phi_n)(P)=(\con_{n=1}^{\varkappa(p,q)} \phi_n)(P)$
 для $P\in \mathcal{P}(^{p,q}X)$. У випадку, коли цей функціонал
 неперервний, його можна продовжити до лінійного неперервного
 мультиплікативного  функціонала $\phi$ на $\W(X)$. В
 цьому випадку будемо використовувати те саме позначення $\phi=\con_{n=1}^\infty
 \phi_n$.

 \begin{theorem} \label{thm8.4}
  Існує послідовність спряжених банахових просторів $\{E_n\}_{n=1}^\infty$
  і по\-слі\-дов\-ність відображень $\delta^{(n)}:E_n \to M_{\W(X)}$
  таких, що довільний характер $\phi \in M_{\W(X)}$ можна
  подати у вигляді $\phi = \con_{n=1}^\infty
  \delta^{(n)}(u_n)$ для деяких елементів $u_n \in E_n$, $n=1,2,\ldots$.
 \end{theorem}
 \begin{proof}
  Покладемо $E_1 = \mathcal{P}(^{0,1}X)'=\overline{X}''$, $\delta^{(1)}=\widetilde{\delta}$
  --- функціонал значення функції в точці $\overline{X}''$.
  Припустимо, що простори $E_k$ і відображення $\delta^{(k)}$ вже
  побудовано для $k<n$. Позначимо $E_n$ множину $\{\pi_{\gamma(n)}(\phi):\quad \phi \in
  \Phi_{n-1}\}$, де $\pi_{\gamma(n)}(\phi)$ ---
  звуження $\phi$ на підпростір $\mathcal{P}(^{\gamma(n)}X)$.
  Очевидно, кожен елемент множини $E_n$ є лінійним функціоналом на
  просторі $\mathcal{P}(^{\gamma(n)}X)$, який перетворюється в
  нуль на всіх елементах простору $\mathcal{P}(^{\gamma(n)}X)\cap
  \W_{\Lambda_{\gamma(n)}}(X)$. Отже, кожен елемент множини $E_n$
  є елементом простору $\mathcal{P}(^{\gamma(n)}X)' \cap
  I_{n-1}^\bot$, де через $I_{n-1}^\bot$ позначено простір
  лінійних функціоналів на $\W(X)$, які перетворюються в нуль на
  ідеалі $I_{n-1}$. Якщо $\W_{\Lambda_{\gamma(n)}}(X) = \W_{\overline{\Lambda}_{\gamma(n)}}(X)$
  то $E_n=\{0\}$, інакше за теоремою \ref{thm8.1} множина $E_n$
  містить ненульові точки.

  Покажемо, що $E_n$ є лінійним простором зі звичайними операціями
  суми елементів і множення на скаляр. Для цього достатньо
  показати, що ці операції не виводять за межі $E_n$.

  Нехай $\phi, \psi \in
  \Phi_{n-1}$.
  Для довільного $P \in \mathcal{P}(^{\gamma(x)}X)$ за теоремою
  \ref{thm8.2} $(\phi*\psi)(P)=\phi(P)+\psi(P)$. Звідси
  $\pi_{\gamma(n)}(\phi*\psi)(P)=\pi_{\gamma(n)}(\phi)(P)+\pi_{\gamma(n)}(\psi)(P)$.
  Отже,
  $\pi_{\gamma(n)}(\phi)+\pi_{\gamma(n)}(\psi)=\pi_{\gamma(n)}(\phi*\psi)$.

  Нехай $\phi \in \Phi_{n-1}$, $a$ --- довільне комплексне число.
  Покажемо, що $a\pi_{\gamma(n)}(\phi) \in E_n$. Оскільки $a\phi \in
  \W(X)'$ і $a\phi$ перетворюється в нуль на елементах простору $\mathcal{P}(^{\gamma(n)}X)
  \cap \W_{\Lambda_{\gamma(n)}}(X)$, то за теоремою \ref{thm5.1}
  існує такий характер $\psi$, що
  $\pi_{\gamma(n)}(\psi)=\pi_{\gamma(n)}(a\phi)$. Звідси, $a\pi_{\gamma(n)}(\phi)=
  \pi_{\gamma(n)}(a\phi)=\pi_{\gamma(n)}(\psi) \in E_n$, оскільки
  $\psi \in \Phi_{n-1}$.
  Отже, $E_n$ --- лінійний простір.

  Визначимо простір $W_n = \mathcal{P}(^{\gamma(n)}X)/(I_{n-1} \cap
  \mathcal{P}(^{\gamma(n)}X))$. Тоді $W_n$ є банаховим простором
  лінійних функціоналів на просторі $E_n$. Визначимо норму $\|\cdot\|_n$ на
  $E_n$ як супремум значень на векторі із $E_n$
  елементів одиничної кулі простору $W_n$. Очевидно, простір
  $W_n'$ співпадає з простором $\mathcal{P}(^{\gamma(n)}X)' \cap
  I_{n-1}^\bot$. Як було вже сказано, кожен елемент простору $E_n$
  міститься в $\mathcal{P}(^{\gamma(n)}X)' \cap
  I_{n-1}^\bot$. З іншого боку, якщо $u\in\mathcal{P}(^{\gamma(n)}X)' \cap
  I_{n-1}^\bot$, то за теоремою \ref{thm5.1} існує такий характер $\phi \in
\Phi_{n-1}$, що $u=\pi_{\gamma(n)}(\phi)$, отже, $u \in E_n$. Звідси
  робимо висновок, що $E_n=W_n'$.

  Для $w \in E_n$ нехай $\delta^{(n)}(w)$ --- характер, який
  існує за теоремою \ref{thm5.1}, якщо в ролі лінійного функціонала взяти $w$.
  Для довільного характера $\phi$ покладемо
  $u_1=\pi_{\gamma(1)}(\phi)\in \overline{X}''$. Припустимо, що уже
  визначено елементи $u_k \in E_k$ для $k<n$. Покладемо
  $$
   u_n = \pi_{\gamma(n)}(\phi)-\pi_{\gamma(n)}\left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right).
  $$
  Покажемо, що $u_n \in E_n$. Достатньо перевірити, що для кожного
  $(p,q)$-полінома $P \in \mathcal{P}(^{\gamma(n)}X)$, який є
  добутком $(p,q)$-поліномів $P_1, P_2$ таких, що $(\deg_1 P_1,\deg_2 P_1 )=(n_1,l_1)\neq(0,0)$
   і $(\deg_1 P_2,\deg_2 P_2 )=(n_2,l_2)\neq(0,0)$ буде
   $u_n(P)=0$. Із мультиплікативності $\phi$ і
   $\con_{k=1}^{n-1}\delta^{(k)}(u_k)$ отримуємо
  \begin{multline*}
   u_n(P) = \pi_{\gamma(n)}(\phi)(P_1 P_2)-\pi_{\gamma(n)}\left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_1 P_2)
   = \phi(P_1 P_2)-\left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_1 P_2)
    \\
   = \phi(P_1) \phi(P_2)-\left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_1) \left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_2)
    \\
   = \pi_{n_1,l_1}(\phi)(P_1) \pi_{n_2,l_2}(\phi)(P_2)-\left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_1) \left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_2).
  \end{multline*}

  За теоремою \ref{thm8.3}
  $$
  \left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_1)
  = \left(\con_{k=1}^{\varkappa(n_1,l_1)}
   \delta^{(k)}(u_k)\right)(P_1)
  $$
  і
  $$
  \left(\con_{k=1}^{n-1}
   \delta^{(k)}(u_k)\right)(P_2)
  = \left(\con_{k=1}^{\varkappa(n_2,l_2)}
   \delta^{(k)}(u_k)\right)(P_2).
  $$

  Згідно із припущенням
  $$
   u_{\varkappa(n_1,l_1)}(P_1)=\pi_{n_1,l_1}(\phi)(P_1)-\left(\con_{k=1}^{\varkappa(n_1,l_1)-1}
   \delta^{(k)}(u_k)\right)(P_1)
  $$
  і
  $$
   u_{\varkappa(n_2,l_2)}(P_2)=\pi_{n_2,l_2}(\phi)(P_2)-\left(\con_{k=1}^{\varkappa(n_2,l_2)-1}
   \delta^{(k)}(u_k)\right)(P_2).
  $$
  Тому, скориставшись теоремою \ref{thm8.2}, маємо
  \begin{multline*}
   u_n(P)=
    \left(
     u_{\varkappa(n_1,l_1)}(P_1)+
     \left(
      \con_{k=1}^{\varkappa(n_1,l_1)-1}\delta^{(k)}(u_k)
     \right)(P_1)
    \right)
     \\ \times
    \left(
     u_{\varkappa(n_2,l_2)}(P_2)+
     \left(
      \con_{k=1}^{\varkappa(n_2,l_2)-1}\delta^{(k)}(u_k)
     \right)(P_2)
    \right)
     \\
    -
    \left(\con_{k=1}^{\varkappa(n_1,l_1)}
   \delta^{(k)}(u_k)\right)(P_1)
   \left(\con_{k=1}^{\varkappa(n_2,l_2)}
   \delta^{(k)}(u_k)\right)(P_2)
    \\
   =
    \left(\con_{k=1}^{\varkappa(n_1,l_1)}
   \delta^{(k)}(u_k)\right)(P_1)
   \left(\con_{k=1}^{\varkappa(n_2,l_2)}
   \delta^{(k)}(u_k)\right)(P_2)
    \\
   -
    \left(\con_{k=1}^{\varkappa(n_1,l_1)}
   \delta^{(k)}(u_k)\right)(P_1)
   \left(\con_{k=1}^{\varkappa(n_2,l_2)}
   \delta^{(k)}(u_k)\right)(P_2)
   = 0.
  \end{multline*}
 Розглянемо функціонал $\con_{j=1}^\infty \delta^{(j)}(u_j)$.
 Нехай $f=\sum_{m=0}^s \sum_{k=0}^m f_{k,m-k}$ --- довільна функція,
 породжена скінченними лінійними комбінаціями і добутками $(p,q)$-поліномів.
 Оскільки $u_k \in E_k$, то за теоремою \ref{thm8.3} маємо
 \begin{multline*}
 \left(
 \con_{j=1}^\infty \delta^{(j)}(u_j)
 \right)(f)
 =
 \sum_{m=0}^s \sum_{k=0}^m
 \left(
 \con_{j=1}^\infty \delta^{(j)}(u_j)
 \right)(f_{k,m-k})
  \\ =
 f(0)+
  \sum_{m=1}^s \sum_{k=0}^m
 \left(
 \con_{j=1}^{\varkappa(k,m-k)} \delta^{(j)}(u_j)
 \right)(f_{k,m-k}).
 \end{multline*}
  Отже, згортка $\con_{j=1}^\infty \delta^{(j)}(u_j)$ є визначеною
 на алгебрі функцій, породжених скінченними лінійними комбінаціями
 і добутками $(p,q)$-поліномів. З іншого боку, для
 довільного $P\in\mathcal{P}(^{k,l}X)$
 \begin{multline*}
  \left(
   \phi-\con_{j=1}^\infty \delta^{(j)}(u_j)
  \right)(P)
  =
   \pi_{k,l}(\phi)(P)-
   \left(
    \con_{j=1}^{\varkappa(k,l)} \delta^{(j)}(u_j)
   \right)(P)
  \\
  =
   u_{\varkappa(k,l)}(P)+
   \left(
    \con_{j=1}^{\varkappa(k,l)-1} \delta^{(j)}(u_j)
   \right)(P)
   -
   \left(
    \con_{j=1}^{\varkappa(k,l)} \delta^{(j)}(u_j)
   \right)(P)
    \\
   =
    \delta^{(\varkappa(k,l))}(u_{\varkappa(k,l)})(P)+
   \left(
    \con_{j=1}^{\varkappa(k,l)-1} \delta^{(j)}(u_j)
   \right)(P)
   -
   \left(
    \con_{j=1}^{\varkappa(k,l)} \delta^{(j)}(u_j)
   \right)(P)=0.
 \end{multline*}
  Отже, $\phi$ співпадає з $\con_{j=1}^\infty \delta^{(j)}(u_j)$
 на алгебрі функцій, породжених скінченними лінійними комбінаціями
 і добутками $(p,q)$-поліномів. Ця алгебра є щільною в
 $\W(X)$, тому $\phi=\con_{j=1}^\infty \delta^{(j)}(u_j)$ на
 $\W(X)$.
 \end{proof}

 \section{Диференціювання на алгебрі $\W(X)$}
 Нехай $u_{p,q} \in E_{\varkappa(p,q)}$. Визначимо лінійний
 функціонал $\theta(u_{p,q})$ на алгебрі $\W(X)$ фор\-му\-лою
 $$
  \theta(u_{p,q})(f)=
   \left\{
    \begin{array}{ll}
     u_{p,q}(f), & \textrm{якщо } f \in \mathcal{P}(^{p,q}X), \\
     0,          & \textrm{якщо } f \in \mathcal{P}(^{m,k}X)
\textrm{ і } (m,k)\neq (p,q).
    \end{array}
   \right.
 $$
 Визначимо лінійний оператор $\partial_{(p,q)}(u_{p,q})$ на $\W(X)$
 формулою
 $$
  \partial_{(p,q)}(u_{p,q})(f)(x)=\theta(u_{p,q})(y \mapsto
  (T_xf)(y)).
 $$

 Нехай $n\geq p$ і  $l\geq q$ і нехай $A_P$ --- це $(n,l)$-лінійна симетрична
 форма, асоційована з деяким $(n,l)$-по\-лі\-но\-мом $P$. Тоді
 відображення $y\mapsto A_P(x^{n-p},y^p; x^{l-q},y^q)$ буде
 $(p,q)$-по\-лі\-но\-мом. Значення $u_{p,q}(y\mapsto A_P(x^{n-p},y^p; x^{l-q},y^q))$
 будемо позначати $\widehat{A}_P(x^{n-p};x^{l-q},u_{p,q})$.
  \begin{theorem}
  Нехай $u_{p,q} \in E_{\varkappa(p,q)}$. Тоді оператор
  $\partial_{(p,q)}(u_{p,q})$ є неперервним оператором
  диференціювання на алгебрі $\W(X)$. При цьому для $P \in \mathcal{P}(^{n,l}X)$
  \begin{equation} \label{eq9.1}
   \partial_{(p,q)}(u_{p,q})(P)(x)=
   \left\{
    \begin{array}{ll}
      C_n^p
      C_l^q
      \widehat{A}_P(x^{n-p};x^{l-q},u_{p,q}), & \textrm{якщо
      $n\geq p$  і  $l\geq q$,} \\
      0, & \textrm{в іншому випадку.}
     \end{array}
   \right.
  \end{equation}
 Також для кожної функції $f \in \W(X)$, $f=\sum_{m=0}^\infty \sum_{k=0}^m
 f_{k,m-k}$,
 $$
  \delta^{(\varkappa(p,q))}(u_{p,q})(f)=
  \sum_{n=1}^\infty \frac{(p!)^n}{(np)!} \frac{(q!)^n}{(n q)!}
  \partial_{(p,q)}^n (u_{p,q}) (f_{np,nq}).
 $$
 \end{theorem}
 \begin{proof}
  Нехай $P\in \mathcal{P}(^{n,l}X)$. Тоді
  \begin{multline*}
   \partial_{(p,q)}(u_{p,q})(P)(x)=\theta(u_k)(y\mapsto
   (T_xP)(y))=\theta(u_k)(y\mapsto
   P(x+y)) \\
   = \theta(u_{p,q})(y\mapsto \sum_{i=0}^n \sum_{j=0}^l
   C_n^i
   C_l^j
   A_P(x^{n-i}, y^i;x^{l-j},y^j)
   )\\=\sum_{i=0}^n \sum_{j=0}^l
   C_n^i
   C_l^j
   \theta(u_{p,q})(y\mapsto
   A_P(x^{n-i}, y^i;x^{l-j},y^j)
   ).
  \end{multline*}
Оскільки
 \begin{multline*}
  \theta(u_{p,q})(y\mapsto A_P(x^{n-i},y^i;x^{l-j},y^j))\\=
   \left\{
    \begin{array}{ll}
     u_{p,q}(y\mapsto A_P(x^{n-i},y^i;x^{l-j},y^j)), & \textrm{якщо } i=p \textrm{ і } j=q, \\
     0,          & \textrm{в іншому випадку,}
    \end{array}
   \right.
 \end{multline*}
 то
  $$
   \partial_{(p,q)}(u_{p,q})(P)(x)=
   \left\{
    \begin{array}{ll}
      C_n^p C_l^q
      \widehat{A}_P(x^{n-p};x^{l-q},u_{p,q}), & \textrm{якщо }
      n\geq p \textrm{ і } l\geq q, \\
      0, & \textrm{в іншому випадку.}
     \end{array}
   \right.
  $$
  Отже, рівність (\ref{eq9.1}) доведено.

  Доведемо, що для кожної функції $f$ із $\W(X)$ функція
  $g(x)=\partial_{(p,q)}(u_{p,q})(f)(x)$ буде належати алгебрі
  $\W(X)$. Нехай $f=\sum_{m=0}^\infty \sum_{k=0}^m f_{k,m-k}$.
  Тоді
  \begin{multline*}
  g(x) = \sum_{m=0}^\infty \sum_{k=0}^m
  \partial_{(p,q)}(u_{p,q})(f_{k,m-k})(x)  \\ =
  \sum_{m=p+q}^\infty \sum_{k=p}^{m-q} C_k^p C_{m-k}^q u_{p,q}
  (y \mapsto A_{f_{k,m-k}}(x^{k-p}, y^p; x^{m-k-q}, y^q)).
  \end{multline*}
 Позначимо
 $
 g_{k-p,m-k-q}(x) = C_k^p C_{m-k}^q u_{p,q}
  (y \mapsto A_{f_{k,m-k}}(x^{k-p}, y^p; x^{m-k-q}, y^q)).
 $

Тоді $g = \sum_{m=p+q}^\infty \sum_{k=p}^{m-q} g_{k-p, m-k-q}$ і
 $g_{k-p,m-k-q} \in \mathcal{P}(^{k-p,m-k-q}X)$.
 Оцінимо норму\linebreak $(k-p,m-k-q)$-полінома $g_{k-p,m-k-q}$.

\vskip-1cm
 \begin{multline*}
 \|g_{k-p,m-k-q}\|\! =\!\! \sup_{\|x\| \leq 1} |g_{k-p,m-k-q}(x)|
  \!=\!\! \sup_{\|x\| \leq 1} |C_k^p C_{m-k}^q u_{p,q}
  (y \mapsto A_{f_{k,m-k}}(x^{k-p}, y^p; x^{m-k-q}, y^q))|
 \\
 \leq \sup_{\|x\| \leq 1} C_k^p C_{m-k}^q \|u_{p,q}\|
 \|y \mapsto A_{f_{k,m-k}}(x^{k-p}, y^p; x^{m-k-q}, y^q)\|
 \\
 = \sup_{\|x\| \leq 1} C_k^p C_{m-k}^q \|u_{p,q}\|
 \sup_{\|y\| \leq 1} |A_{f_{k,m-k}}(x^{k-p}, y^p; x^{m-k-q},
 y^q)|
  \\
 \leq \sup_{\|x\| \leq 1} C_k^p C_{m-k}^q \|u_{p,q}\|
 \sup_{\|y\| \leq 1} \|A_{f_{k,m-k}}\| \|x\|^{k-p} \|y\|^p
 \|x\|^{m-k-q} \|y\|^q
 = C_k^p C_{m-k}^q \|u_{p,q}\| \|A_{f_{k,m-k}}\|.
 \end{multline*}
 Оскільки $C_k^p \leq 2^k$, $C_{m-k}^q \leq 2^{m-k}$ і
  $\|A_{f_{k,m-k}}\| \leq (2e)^m \|f_{k,m-k}\|$, то
 $$
 \| g_{k-p,m-k-q} \| \leq 2^k 2^{m-k} (2e)^m  \|u_{p,q}\|\|f_{k,m-k}\|=
 (4e)^m \|u_{p,q}\| \|f_{k,m-k}\|.
 $$
 Нехай $r>0$. Має місце оцінка
 $$
 \|g\|_r = \sum_{m=p+q}^\infty \sum_{k=p}^{m-q} r^{m-p-q}
 \| g_{k-p,m-k-q} \| \leq \|u_{p,q}\| \sum_{m=p+q}^\infty \sum_{k=p}^{m-q} r^{m-p-q}  (4e)^m \|f_{k,m-k}\|.
 $$
 Позначимо $\rho = \max\{1, r\}$. Тоді $r^{m-p-q}\leq \rho^{m-p-q} \leq
 \rho^m$. Тому
 $$
 \|g\|_r \leq \|u_{p,q}\|\sum_{m=p+q}^\infty \sum_{k=p}^{m-q} \rho^m (4e)^m
 \|f_{k,m-k}\| \leq  \|u_{p,q}\| \sum_{m=0}^\infty \sum_{k=0}^m (4 \rho e)^m
 \|f_{k,m-k}\| = \|u_{p,q}\| \|f\|_{4\rho e}.
 $$
 Оскільки $f \in \W(X)$, то $\|f\|_{4\rho e} < +\infty$. Тому $\|g\|_r <
 +\infty$. Отже, $g \in \W(X)$.

 Нехай $P\in \mathcal{P}(^{n,l}X)$, $Q \in \mathcal{P}(^{m,r} X)$.
 Доведемо, що
 \begin{equation} \label{eq9.2}
  \partial_{(p,q)}(u_{p,q})(PQ) = P \partial_{(p,q)}(u_{p,q})(Q) +
  Q \partial_{(p,q)}(u_{p,q})(P).
 \end{equation}
 Якщо $n+m<p$ або $l+r<q$, то рівність очевидна. Нехай $n+m \geq
 p$ і $l+r \geq q$. Нехай $A_{PQ}$ --- форма, асоційована з
 $(n+m,l+r)$-поліномом $PQ$. Із рівності (\ref{eq9.1}) випливає,
 що
 $$
 \partial_{(p,q)}(u_{p,q})(PQ)(x) = C_{n+m}^p
 C_{l+r}^q
 \theta(u_{p,q})(y \mapsto A_{PQ}(x^{n+m-p}, y^p; x^{l+r-q},
 y^q)).
 $$
 Нехай $z_1,\ldots,z_{n+m},t_1,\ldots,t_{l+r} \in X$. Тоді
 \begin{multline} \label{eq9.3}
  A_{PQ}(z_1,\ldots,z_{n+m};t_1,\ldots,t_{l+r}) =
  \frac{1}{(n+m)!} \frac{1}{(l+r)!}\\ \times \sum_{\sigma \in S(n+m)}
  \sum_{\tau \in S(l+r)}
  A_P(z_{\sigma(1)}, \ldots, z_{\sigma(n)}; t_{\tau(1)}, \ldots, t_{\tau(l)})
  \\ \times
  A_Q(z_{\sigma(n+1)}, \ldots, z_{\sigma(n+m)}; t_{\tau(l+1)}, \ldots,
  t_{\tau(l+r)}).
 \end{multline}
 Введемо позначення
 $
 z=(z_1,\ldots,z_{n+m})$,
 $t=(t_1,\ldots,t_{l+r}),
 $
 $
 \sigma_1 = \sigma|_{\{1, \ldots, n\}}$,
 $\sigma_2 = \sigma|_{\{n+1, \ldots, n+m\}}$,
 $
 \tau_1 = \tau|_{\{1, \ldots, l\}}$,
  $\tau_2 = \tau|_{\{l+1, \ldots, l+r\}}.
 $
 Розглянувши $z$ як відображення з $\{1,\ldots,n+m\}$ в $X$, а $t$
 як відображення з $\{1,\ldots,l+r\}$ в $X$, формулу (\ref{eq9.3})
 перепишемо у вигляді
 \begin{multline}
  A_{PQ}(z;t) = \frac{1}{(n+m)!} \frac{1}{(l+r)!} \sum_{\sigma \in S(n+m)}
  \sum_{\tau \in S(l+r)}
  A_P(z \circ \sigma_1; t \circ \tau_1) A_Q(z \circ \sigma_2; t \circ
  \tau_2).
 \end{multline}
 Нехай
 $$
  a(\sigma) = \bigl|\bigl\{1,\ldots,n\bigr\} \cap \sigma^{-1}\bigl\{n+m-p+1, \ldots,
  n+m\bigr\}\bigr|,
 $$
 $$
  b(\tau) = \bigl|\bigl\{1,\ldots,l\bigr\} \cap \tau^{-1}\bigl\{l+r-q+1, \ldots,
  l+r\bigr\}\bigr|.
 $$
 Тоді
 $$
  A_P(z \circ \sigma_1; t \circ \tau_1) = A_P(x^{n-a(\sigma)}, y^{a(\sigma)}; x^{l-b(\tau)},
  y^{b(\tau)}),
 $$
 $$
  A_Q(z \circ \sigma_2; t \circ \tau_2) = A_Q(x^{m-p+a(\sigma)}, y^{p-a(\sigma)}; x^{r-q+b(\tau)},
  y^{q-b(\tau)}).
 $$
 Тепер
 \begin{multline*}
  \partial_{(p,q)}(u_{p,q})(PQ)(x)
  = C_{n+m}^p
    C_{l+r}^q
    \frac{1}{(n+m)!}\frac{1}{(l+r)!}
    \sum_{\sigma \in S(n+m)} \sum_{\tau \in S(l+r)}
    \theta(u_{p,q})(y
    \\
    \mapsto
    A_P(x^{n-a(\sigma)}, y^{a(\sigma)}; x^{l-b(\tau)}, y^{b(\tau)})
    A_Q(x^{m-p+a(\sigma)}, y^{p-a(\sigma)}; x^{r-q+b(\tau)}, y^{q-b(\tau)})
    ).
 \end{multline*}
 Оскільки $u_{p,q} \in E_{\varkappa(p,q)}$, то в останньому виразі
 не дорівнюють нулю тільки ті доданки, для яких $a(\sigma)=p$ і
 $b(\tau)=q$ або $a(\sigma)=b(\tau)=0$. Кількість тих $\sigma \in
 S(n+m)$, для яких $a(\sigma)=p$, дорівнює
 $$
 \bigl|\bigl\{\sigma \in S(n+m): \quad a(\sigma)=p\bigl\}\bigl| =
 \left\{
  \begin{array}{ll}
   C_n^p p! (n+m-p)!, & \textrm{при
   }n \geq p, \\
  0, & \textrm{при } n<p.
  \end{array}
 \right.
 $$
 Аналогічно
 $$
 \bigl|\bigl\{\tau \in S(l+r): \quad b(\tau)=q\bigr\}\bigr| =
 \left\{
  \begin{array}{ll}
   C_l^q q! (l+r-q)!, & \textrm{при
   }l \geq q, \\
  0, & \textrm{при } l<q,
  \end{array}
 \right.
 $$
 $$
 \bigl|\bigl\{\sigma \in S(n+m): \quad a(\sigma)=0\bigr\}\bigr| =
 \left\{
  \begin{array}{ll}
   C_m^p p! (n+m-p)!, & \textrm{при
   }m \geq p, \\
    0, & \textrm{при } m<p,
  \end{array}
 \right.
 $$
 $$
 \bigl|\bigl\{\tau \in S(l+r): \quad b(\tau)=0\bigr\}\bigr| =
 \left\{
  \begin{array}{ll}
   C_r^q q! (l+r-q)!, & \textrm{при
   }r \geq q, \\
   0, & \textrm{при } r<q,
  \end{array}
 \right.
 $$
 Звідси для $n \geq p$, $m \geq p$, $l \geq q$, $r \geq q$
 \begin{multline*}
 \partial_{(p,q)}(u_{p,q})(PQ)(x) =
   C_{n+m}^p\;
   C_{l+r}^q\;
 \frac{1}{(n+m)!} \frac{1}{(l+r)!}
 p!\;(n+m-p)!\;
 q!\;(l+r-q)!\;
 \\
 \times
 \Bigl(
  C_n^p\;
  C_l^q\;
  u_{p,q}\bigl(y \mapsto A_P(x^{n-p}, y^p; x^{l-q}, y^q) A_Q(x^m; x^r)\bigr)
  \\
   +
  C_m^p\;  C_r^q\;  u_{p,q}\bigl(y\mapsto
  A_P(x^n; x^l) A_Q(x^{m-p}, y^p; x^{r-q}, y^q)\bigr)
 \Bigr).
 \end{multline*}
 Оскільки
 $
 C_{n+m}^p\;
   C_{l+r}^q\;
 \frac{1}{(n+m)!} \frac{1}{(l+r)!}
 p!\;(n+m-p)!\;
 q!\;(l+r-q)!\;
 =1,
 $
 то

  \begin{multline*}
 \partial_{(p,q)}(u_{p,q})(PQ)(x) =
 \Bigl(
 Q(x)\;
  C_n^p\;
  C_l^q\;
  u_{p,q}\bigl(y \mapsto A_P(x^{n-p}, y^p; x^{l-q}, y^q) \bigr)
  \\
   +
   P(x)\;
  C_m^p\;  C_r^q\;  u_{p,q}\bigl(y\mapsto
   A_Q(x^{m-p}, y^p; x^{r-q}, y^q)\bigr)
 \Bigr).
 \end{multline*}
 Враховуючи формулу (\ref{eq9.1}), отримуємо
 $
   \partial_{(p,q)}(u_{p,q})(PQ) = P \partial_{(p,q)}(u_{p,q})(Q) +
  Q \partial_{(p,q)}(u_{p,q})(P).
 $
 Отже, ми довели рівність (\ref{eq9.2}) для випадку $n,m \geq p$ і
 $l,r\geq q$. Для всіх інших випадків цю рівність також перевірити
 нескладно.

 Із рівності (\ref{eq9.2}) і лінійності оператора
 $\partial_{(p,q)}(u_{p,q})$
 випливає, що цей оператор є опе\-ра\-то\-ром диференціювання на алгебрі
 $\W(X)$.

 Нехай $P \in \mathcal{P}(^{np,nq}X)$. Після $n$-кратного
 застосування формули (\ref{eq9.1}) отримаємо
 \begin{multline*}
 \partial_{(p,q)}^n (u_{p,q})(P) = C_{np}^p C_{(n-1)p}^p \ldots
 C_p^p C_{nq}^q C_{(n-1)q}^q \ldots C_q^q
 \delta^{(\varkappa(p,q))}(u_{p,q})(P)  \\ = \frac{(np)!}{(p!)^n}
 \frac{(n q)!}{(q!)^n} \delta^{(\varkappa(p,q))}(u_{p,q})(P).
 \end{multline*}
 Звідси
 $$
  \delta^{(\varkappa(p,q))}(u_{p,q})(P) = \frac{(p!)^n}{(np)!} \frac{(q!)^n}{(n q)!}
  \partial_{(p,q)}^n (u_{p,q})(P).
 $$
 Нехай $f \in \W(X)$, $f = \sum_{m=0}^\infty \sum_{k=0}^m
 f_{k,m-k}$. Тоді
 \begin{multline*}
  \delta^{(\varkappa(p,q))}(u_{p,q})(f) = \sum_{m=0}^\infty
  \sum_{k=0}^m \delta^{(\varkappa(p,q))}(u_{p,q})(f_{k,m-k})  \\
  =
  \sum_{n=1}^\infty \delta^{(\varkappa(p,q))}(u_{p,q})(f_{np, nq})
  = \sum_{n=1}^\infty \frac{(p!)^n}{(np)!} \frac{(q!)^n}{(n q)!}
  \partial_{(p,q)}^n (u_{p,q}) (f_{np,nq}).
 \end{multline*}
 \end{proof}

 \section{Похідна за напрямком у дійсному сенсі на алгебрі $\W(X)$}
 Нехай $h \in X$. Знайдемо похідну за напрямком $h$ від $P \in
 \mathcal{P}(^{p,q}X)$, розглядаючи $X$ як простір з дійсною
 структурою.
 \begin{multline*}
 \lim_{t\in \mathbb{R},\; t\to 0} \frac{P(x+th)-P(x)}{t}
 \\
 =
 \lim_{t\in \mathbb{R},\; t\to 0} \frac{1}{t} \left(
 \sum_{i=0}^p \sum_{j=0}^q C_p^i C_q^j A_P(x^{p-i}, (t h)^i; x^{q-j}, (t h)^j) -
 A_P(x^p;x^q)\right)
 \\
 = \lim_{t\in \mathbb{R},\; t\to 0} \frac{1}{t} \left(
 \sum_{i=0}^p \sum_{j=0}^q C_p^i C_q^j t^{i+j} A_P(x^{p-i}, h^i; x^{q-j},  h^j) -
 A_P(x^p;x^q)\right)
 \\
 = C_p^1 C_q^0 A_P(x^{p-1},h; x^q)+
 C_p^0 C_q^1 A_P(x^p; x^{q-1}, h)
  =
 \partial_{(1,0)}(h_{1,0})(P)(x) +
 \partial_{(0,1)}(h_{0,1})(P)(x),
 \end{multline*}
 де через $h_{1,0}$ позначено лінійний функціонал на просторі
 $\mathcal{P}(^{1,0}X)$, який кожному $g \in \mathcal{P}(^{1,0}X)$
 ставить у відповідність число $g(h)$, через $h_{0,1}$ позначено
 лінійний функціонал на просторі
 $\mathcal{P}(^{0,1}X)$, який кожному $g \in \mathcal{P}(^{0,1}X)$
 ставить у відповідність число $g(h)$.
 Отже,
 $$
 \lim_{t\in \mathbb{R},\; t\to 0} \frac{P(x+th)-P(x)}{t}
 =(\partial_{(1,0)}(h_{1,0})+\partial_{(0,1)}(h_{0,1}))(P)(x).
 $$

 Для $f \in \W(X)$ будемо мати
 \begin{multline*}
  \lim_{t\in \mathbb{R},\; t\to 0}
  \frac{f(x+th)-f(x)}{t}
  =
  \lim_{t\in \mathbb{R},\; t\to 0} \frac{1}{t}
  \left(
  \sum_{m=0}^\infty \sum_{k=0}^m f_{k,m-k}(x+th)-
  \sum_{m=0}^\infty \sum_{k=0}^m f_{k,m-k}(x)
  \right)
  \\
  =
  \sum_{m=0}^\infty \sum_{k=0}^m
  \lim_{t\in \mathbb{R},\; t\to 0}
  \frac{f_{k,m-k}(x+th)-f_{k,m-k}(x)}{t}
  =
  \sum_{m=0}^\infty \sum_{k=0}^m
  (\partial_{(1,0)}(h_{1,0})+\partial_{(0,1)}(h_{0,1}))
  (f_{k,m-k})(x)
  \\
  =
  (\partial_{(1,0)}(h_{1,0})+\partial_{(0,1)}(h_{0,1}))
  (f)(x).
 \end{multline*}
